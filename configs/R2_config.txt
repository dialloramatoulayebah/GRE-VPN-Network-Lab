! R2 Router Configuration - Montreal
! GRE VPN Network Lab
!
hostname R2-Montreal
!
! Interface Configuration
interface GigabitEthernet0/0
 description WAN Interface to ISP
 ip address 200.2.2.2 255.255.255.252
 no shutdown
!
interface GigabitEthernet0/1
 description LAN Interface Montreal
 ip address 192.168.20.1 255.255.255.0
 no shutdown
!
! GRE Tunnel Configuration
interface Tunnel0
 description GRE Tunnel to R1-Toronto
 ip address 10.0.0.2 255.255.255.252
 tunnel source GigabitEthernet0/0
 tunnel destination 200.1.1.1
 tunnel mode gre ip
 no shutdown
!
! EIGRP Configuration
router eigrp 100
 network 192.168.20.0 0.0.0.255
 network 10.0.0.0 0.0.0.3
 no auto-summary
!
! Default Route to ISP
ip route 0.0.0.0 0.0.0.0 200.2.2.1
!
end
